	CLASS ENCLOSURE

	ATTRIBUTES
	 ID : NUMBER
	 NAME : TEXT
	 TEMPERATURE : NUMBER
	 ANIMALS : LIST<ANIMAL>

	CONSTRUCTOR(N:TEXT,T:NUMBER)
	    SET NAME TO N
	    SET TEMPERATURE TO T
	    SET ANIMALS TO EMPTY LIST
	END CONSTRUCTOR

	METHODS

	ADD_ANIMAL(A:ANIMAL)
	    ADD A TO ANIMALS
	END METHOD

	GET_NAME()
	    DISPLAY NAME
	END METHOD

	GET_TEMPERATURE()
	    DISPLAY TEMPERATURE
	END METHOD

	DISPLAY_ANIMALS()
	    REPEAT FROM I = 1 TO SIZE OF ANIMALS
	        DISPLAY ANIMALS[I]
	    END REPEAT
	END METHOD

	END CLASS
