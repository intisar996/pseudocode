 METHODS

 END METHOD

 GET_EXPIREDAY()

CLASS ORDER

ATTRIBUTES
ID : NUMBER
ITEMS : LIST<ITEM>
PAYMENT : PAYMENTMETHOD

CONSTRUCTOR()
    SET ITEMS TO EMPTY LIST
END CONSTRUCTOR

METHODS

ADD_ITEM(I:ITEM)
    ADD I TO ITEMS
END METHOD

SET_PAYMENT(P:PAYMENTMETHOD)
    SET PAYMENT TO P
END METHOD

CHECKOUT()

    SET TOTAL TO 0

    REPEAT FROM I = 1 TO SIZE OF ITEMS
        SET TOTAL TO TOTAL + ITEMS[I].PRICE
    END REPEAT

    SET PAYMENT.AMOUNT TO TOTAL

    PAYMENT.PROCESS_PAYMENT()

END METHOD

END CLASS

AUTHORISE_PURCHASE(AMOUNT:NUMBER)

    IF BALANCE GREATER THAN OR EQUAL TO AMOUNT THEN

CLASS CREDITCARD INHERITS CARD

ATTRIBUTES
CREDITLIMIT : NUMBER
OUTSTANDINGBALANCE : NUMBER

CONSTRUCTOR(CA:NUMBER,EX:TEXT,ST:TEXT,CH:CHIP,CL:NUMBER)
    SUPER(CA,EX,ST,CH)
    SET CREDITLIMIT TO CL
    SET OUTSTANDINGBALANCE TO 0
END CONSTRUCTOR

METHODS

GET_CREDITLIMIT()
    DISPLAY CREDITLIMIT
END METHOD

GET_OUTSTANDINGBALANCE()
    DISPLAY OUTSTANDINGBALANCE
END METHOD

AUTHORISE_PURCHASE(AMOUNT:NUMBER)

    IF OUTSTANDINGBALANCE + AMOUNT LESS THAN OR EQUAL TO CREDITLIMIT THEN
        SET OUTSTANDINGBALANCE TO OUTSTANDINGBALANCE + AMOUNT
        DISPLAY "PURCHASE APPROVED"
    ELSE
        DISPLAY "CREDIT LIMIT EXCEEDED"
    END IF

END METHOD

CLASS STORAGEUNIT
 ATTRIBUTES
  CAPACTIY : NUMBER
  USEDSPACE : NUMBER
 CONSTRUCTOR(CA:NUMBER, US: NUMBER)
  SET CAPACTIY TO CA
  SET USEDSPACE TO US
 END CONSTRUCTOR

METHODS

ADD_RECORDING(SIZE:NUMBER)

    IF USEDSPACE + SIZE LESS THAN OR EQUAL TO CAPACITY THEN
        SET USEDSPACE TO USEDSPACE + SIZE
        DISPLAY "RECORDING SAVED"
    ELSE
        DISPLAY "STORAGE FULL"
    END IF

END METHOD

GET_USEDSPACE()
    DISPLAY USEDSPACE
END METHOD

END CLASS
