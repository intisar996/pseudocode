 CLASS Hospital 
  ATTRIBUTES
   ID : NUMBER
   NAME : TEXT
   ADDRESS : TEXT
   PATIENT : LIST<PATIENT>
  
 CONSTRUCTOR (I : TEXT, N : TEXT , A : TEXT)
    SET ID TO I
    SET NAME TO N
    SET ADDRESS TO 0
    SET PATIENT TO LIST[]
 END CONSTRUCTOR
    
 METHODES
  ADMITPATIENT(P: PATIENT)
   ADD P TO LSIT  PATIENT
  END METHOD 
  
  DUSCHARGEPATIENT(PATIENTID: NUMBER)
  END METHOD 
  
  COUNTPATIENT()
  END METHOD
 
 END CLASS

  CLASS PATIENT
   ATTRIBUTS
    ID : NUMBER
    NAME :  STRING
    DATEOFBRITH : STRING
    ADMISSIONDATE : STRING
    ISADMITTED :BOOLEN
    ROOM : NUMBER
    PROCEDURES: LIST<PROCEDURES>
  
    CONSTRUCTOR(I :NUMBER, N: TEXT, DOB:TEXT, R:NUMBER)
      SET ID TO I
      SET NAME TO N
      SET DATEOFBRITH TO DOB
      SET ISADMITTED TO TRUE
      SET ROOM TO R
      SET PROCEDURES TO LIST
    END CONSTRUCTER
   
   METHODS
     ADDPROCEDURE(PR:PROCEDURES )
     ADD PR TO LIST ADDPROCEDURE
   END METHOD   
    MOVEROOM(R : ROOM)
    SET ROOM TO R
    END METHOD 
    COUNTTOTALCOST()
   END METHOD  
  END CLASS


  CLASS Procedure  
   ATTRIBUTS
    ID : NUMBER
    NAME :  STRING
    performedDate : STRING
    doctorName : STRING
    notes  :STRING
    cost  : NUMBER
  
    CONSTRUCTOR(I :NUMBER, N: TEXT, PRB:TEXT, D:TEXT,NO:TEXT,CO:NUMBER )
      SET ID TO I
      SET NAME TO N
      SET performedDate TO PRB
      SET doctorName TO D
      SET notes TO NO
      SET cost TO CO
    END CONSTRUCTER
   
   METHODS
     DESCRIBE()
     DISPLAY  "NAME" + NAME + "-" "DOCTOR NAME" + doctorName + "DATE" + performedDate     
   END METHOD      
  END CLASS

   CLASS ROOM 
   ATTRIBUTS
    ID : NUMBER
    ROOMNUMBER :  NUMBER
    WARDTYPE : STRING
    FLOOR : NUMBER
    COUNT  :NUMBER
  
    CONSTRUCTOR(I :NUMBER, RNUM: NUMBER, WT:TEXT, F:TEXT,CO:NUMBER )
      SET ID TO I
      SET ROOMNUMBER TO RNUM
      SET WARDTYPE TO WT
      SET FLOOR TO F
      SET COUNT TO CO
    END CONSTRUCTER
           
  END CLASS
