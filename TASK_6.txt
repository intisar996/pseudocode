	TASK 312: Data Center Server Management

	CLASS SERVER

	ATTRIBUTES

	ID : NUMBER
	SERIALNUMBER : TEXT
	UPTIME : NUMBER
	HARDWARESPECIFICATION : HARDWARESPECIFICATION

    CONSTRUCTOR(I:NUMBER,SN:TEXT,UP:NUMBER,HS:HARDWARESPECIFICATION)
		SET ID TO I
		SET SERIALNUMBER TO SN
		SET UPTIME TO UP
		SET HARDWARESPECIFICATION TO HS
    END CONSTRUCTOR

	METHODS

	GET_SERIALNUMBER()
		RETURN SERIALNUMBER
	END METHOD

	GET_UPTIME()
		DISPLAY UPTIME
	END METHOD

	CalculateLoad()
	END METHOD

	END CLASS


	CLASS WEBSERVER INHERITS SERVER

	ATTRIBUTES

	ACTIVECONNECTIONS : NUMBER
	MAXCONNECTIONS : NUMBER

	CONSTRUCTOR(SN:TEXT,UP:NUMBER,HS:HARDWARESPECIFICATION,AC:NUMBER,MC:NUMBER)
		SUPER(SN,UP,HS)
		SET ACTIVECONNECTIONS TO AC
		SET MAXCONNECTIONS TO MC
	END CONSTRUCTOR

	METHODS

     CalculateLoad()
	IF MAXCONNECTIONS GREATER THAN 0 THEN
		RETURN (ACTIVECONNECTIONS * 100) / MAXCONNECTIONS
	ELSE
		RETURN 0
	END IF

	END METHOD

	END CLASS


	CLASS DATABASESERVER INHERITS SERVER

	ATTRIBUTES

	QUERYTHROUGHPUT : NUMBER
	MAXQPS : NUMBER

	CONSTRUCTOR(SN:TEXT,UP:NUMBER,HS:HARDWARESPECIFICATION,QT:NUMBER,MQ:NUMBER)
		SUPER(SN,UP,HS)
		SET QUERYTHROUGHPUT TO QT
		SET MAXQPS TO MQ
	END CONSTRUCTOR

	METHODS

	CalculateLoad()
		DISPLAY (QUERYTHROUGHPUT * 100) / MAXQPS
	END METHOD

	AcceptConnection()
		IF QUERYTHROUGHPUT GREATER THAN OR EQUAL TO MAXQPS THEN
			DISPLAY "CONNECTION REFUSED"
		ELSE
			DISPLAY "CONNECTION ACCEPTED"
		END IF
	END METHOD

	END CLASS


	CLASS CACHESERVER INHERITS DATABASESERVER

	ATTRIBUTES

	CACHEHITRATIO : NUMBER

	CONSTRUCTOR(SN:TEXT,UP:NUMBER,HS:HARDWARESPECIFICATION,QT:NUMBER,MQ:NUMBER,CHR:NUMBER)
		SUPER(SN,UP,HS,QT,MQ)
		SET CACHEHITRATIO TO CHR
	END CONSTRUCTOR

	METHODS

     CalculateLoad()

	SET LOAD TO (QUERYTHROUGHPUT * 100) / MAXQPS
	IF LOAD LESS THAN 0 THEN
		SET LOAD TO 0
	END IF
	RETURN LOAD

	END METHOD
	END CLASS


	CLASS HARDWARESPECIFICATION

	ATTRIBUTES

	ID : NUMBER
	CPU : TEXT
	RAM : TEXT
	STORAGE : TEXT

	CONSTRUCTOR(C:TEXT,R:TEXT,S:TEXT)
		SET CPU TO C
		SET RAM TO R
		SET STORAGE TO S
	END CONSTRUCTOR

	METHODS
     GET_CPU()
	RETURN CPU
	END METHOD

	GET_RAM()
	RETURN RAM
	END METHOD

	GET_STORAGE()
	RETURN STORAGE
	END METHOD
	DisplaySpecification()
		DISPLAY CPU
		DISPLAY RAM
		DISPLAY STORAGE
	END METHOD

	END CLASS


	CLASS RACK

	ATTRIBUTES

	ID : NUMBER
	RACKNAME : TEXT
	SERVERS : LIST<SERVER>

	CONSTRUCTOR(RN:TEXT)
		SET RACKNAME TO RN
		SET SERVERS TO EMPTY LIST
	END CONSTRUCTOR

	METHODS

	AddServer(S:SERVER)
		ADD S TO SERVERS
	END METHOD

		DisplayServers()

		REPEAT FROM I = 1 TO SIZE OF SERVERS

			DISPLAY SERVERS[I].GET_SERIALNUMBER()

		END REPEAT

		END METHOD

	END CLASS


	START
	BEGIN

	CREATE OBJECT HS1 OF SERVER ("Intel Xeon","64GB","2TB SSD")

	CREATE OBJECT WS1 OF WEBSERVER("WEB001",240,HS1,300,500)

	CREATE OBJECT DB1 OF DATABASESERVER("DB001",500,HS1,800,1000)

	CREATE OBJECT CS1 OF CACHESERVER("CACHE001",700,HS1,700,1000,90)

	CREATE OBJECT R1 OF RACK("Rack A")

	R1.AddServer(WS1)
	R1.AddServer(DB1)
	R1.AddServer(CS1)

	WS1.CalculateLoad()
	DB1.CalculateLoad()
	DB1.AcceptConnection()
	CS1.CalculateLoad()

	R1.DisplayServers()

	END
	STOP
