Task 303: Musical Instruments Store


  
CLASS INSTRUMENT
  ATTRIBUTES
   BRAND : TEXT
   PRICE : NUMBER
   
   CONSTRUCTOR(B:TEXT,P:NUMBER)
    SET BRAND TO B
    SET PRICE TO P
   END CONSTRUCTOR

   METHODS

    GET_BRAND()
     DISPLAY BRAND
    END METHOD

    GET_PRICE()
     DISPLAY PRICE
    END METHOD

    DESCRIBEPLAYERSTYLE()
      DISPLAY "HOW IT IS PLAYED"
    END METHOD

END CLASS

 CLASS GUITAR INHERITS INSTRUMENT
   ATTRIBUTES
    FEWSTRING:TEXT

   CONSTRUCTOR(FEW:TEXT,B:TEXT,P:NUMBER)
    SUPER(B,P)
    SET FEWSTRING TO FEW
   END CONSTRUCTOR

   METHODS
    GET_FEWSTRING()
     DISPLAY FEWSTRING
   END METHOD

    OVERRIDE DESCRIBEPLAYERSTYLE()
     DISPLAY "strummed or plucked"
    END METHOD
  
  END CLASS

   CLASS DRUMKIT INHERITS INSTRUMENT
    ATTRIBUTES
     DRUM_PIECES : NUMBER

    CONSTRUCTOR(DRUP:NUMBER,B:TEXT,P:NUMBER)
     SET DRUM_PIECES TO DRUP
      SUPER(B,P)
    END CONSTRUCTOR

    METHODS
     GET_DRUM_PIECES()
      DISPLAY DRUM_PIECES
    END METHOD
 
    OVERRIDE DESCRIBEPLAYERSTYLE()
     DISPLAY "struck"
    END METHOD

    END CLASS

  CLASS KEYBOARD INHERITS INSTRUMENT
   ATTRIBUTES
    SEVERAL_KEYS:TEXT

   CONSTRUCTOR(SRKEY:TEXT,B:TEXT,P:NUMBER)
     SUPER(B,P)
     SET SEVERAL_KEYS TO SRKEY
   END CONSTRUCTOR
    
    METHODS

    GET_SEVERAL_KEYS()
     DISPLAY SEVERAL_KEYS
    END METHOD

    OVERRIDE DESCRIBEPLAYERSTYLE()
     DISPLAY "pressed"
    END METHOD
  END CLASS
   CLASS STORE
   ATTRIBUTES
    INSTRUMENT : LIST<INSTRUMENT>

    CONSTRUCTOR(INST:LIST<INSTRUMENT>)
      SET INSTRUMENT TO INST
    END CONSTRUCTOR

   METHODS
    ADD_INSTRUMENT(ADDINSTRUMENT:INSTRUMENT)
      ADD ADDINSTRUMENT TO INSTRUMENT
   END METHOD

END CLASS

START
 BEGIN
     CREATE OBJECT GUITAR1 OF GUITAR("6 Strings", "Yamaha", 450)
    CREATE OBJECT DRUMKIT1 OF DRUMKIT(5, "Pearl", 900)
    CREATE OBJECT KEYBOARD1 OF KEYBOARD("88 Keys", "Casio", 700)

    CREATE OBJECT STORE1 OF STORE(EMPTY LIST)

    STORE1.ADD_INSTRUMENT(GUITAR1)

CLASS ELECTRICITYBILL INHERITS BILL

  ATTRIBUTES
    UNITS_POWER : NUMBER
    RATE : NUMBER
    TIME : NUMBER


  CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,UNP:NUMBER,RA:NUMBER,TI:NUMBER)
    SUPER(I,BILP,DUDATE)
    SET UNITS_POWER TO UNP
    SET RATE TO RA
    SET TIME TO TI
  END CONSTRUCTOR


  METHODS

    OVERRIDE CALCULATEAMOUNTOWED()
      DISPLAY UNITS_POWER * (RATE * TIME)
    END METHOD


    GET_UNITS_POWER()
      DISPLAY UNITS_POWER
    END METHOD


    GET_RATE()
      DISPLAY RATE
    END METHOD


    GET_TIME()
      DISPLAY TIME
    END METHOD

END CLASS



CLASS WATERBILL INHERITS BILL

  ATTRIBUTES
    LITERS_CONSUMED : NUMBER
    FLAT_RATE : NUMBER
    FIXEDSTANDING : NUMBER


  CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,LI:NUMBER,FRA:NUMBER,FIX:NUMBER)
    SUPER(I,BILP,DUDATE)
    SET LITERS_CONSUMED TO LI
    SET FLAT_RATE TO FRA
    SET FIXEDSTANDING TO FIX
  END CONSTRUCTOR


  METHODS

    OVERRIDE CALCULATEAMOUNTOWED()
      DISPLAY (LITERS_CONSUMED * FLAT_RATE) + FIXEDSTANDING
    END METHOD


    GET_LITERS_CONSUMED()
      DISPLAY LITERS_CONSUMED
    END METHOD


    GET_FLAT_RATE()
      DISPLAY FLAT_RATE
    END METHOD


    GET_FIXEDSTANDING()
      DISPLAY FIXEDSTANDING
    END METHOD

END CLASS



CLASS GASBILL INHERITS BILL

  ATTRIBUTES
    UNITS_CONSUMED : NUMBER
    RATE : NUMBER
    SEASON : TEXT


  CONSTRUCTOR(I:NUMBER,BILP:NUMBER,DUDATE:DATE,UNI:NUMBER,RA:NUMBER,SE:TEXT)
    SUPER(I,BILP,DUDATE)
    SET UNITS_CONSUMED TO UNI
    SET RATE TO RA
    SET SEASON TO SE
  END CONSTRUCTOR


  METHODS

    OVERRIDE CALCULATEAMOUNTOWED()

      IF SEASON EQUAL "WINTER" THEN
        DISPLAY (UNITS_CONSUMED * RATE) + 5
      ELSE
        DISPLAY (UNITS_CONSUMED * RATE)
      END IF

    END METHOD


    GET_UNITS_CONSUMED()
      DISPLAY UNITS_CONSUMED
    END METHOD


    GET_RATE()
      DISPLAY RATE
    END METHOD


    GET_SEASON()
      DISPLAY SEASON
    END METHOD

END CLASS
